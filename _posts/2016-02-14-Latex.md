## latex中文支持 （CJK篇）

%!TEX encoding = UTF-8 Unicode 
\documentclass[12pt]{article} 
\usepackage{CJKutf8} 

\title{TownShip 游戏解析}

\begin{document}
 
\begin{CJK}{UTF8}{bsmi} 
\maketitle
各种中文
\end{CJK} 

\end{document}
